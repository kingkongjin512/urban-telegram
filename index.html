let refImgBase64 = null; // 参考图base64存储

// 参考图预览
function previewRefImg(input) {
  const file = input.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = function(e) {
    refImgBase64 = e.target.result;
    document.getElementById('refPreview').src = refImgBase64;
    document.getElementById('refPreview').style.display = 'block';
  };
  reader.readAsDataURL(file);
}

// 批量生成图片
async function generateImgs() {
  const prompt = document.getElementById('prompt').value;
  const size = document.getElementById('size').value;
  const style = document.getElementById('style').value;
  const color = document.getElementById('color').value;
  const font = document.getElementById('font').value;
  const count = document.getElementById('count').value;
  const resultDiv = document.getElementById('result');

  // 校验必填项
  if (!prompt.trim()) {
    alert('请输入详情图描述！');
    return;
  }

  // 拼接完整提示词（自动整合所有设置）
  let fullPrompt = `${prompt}，电商详情图，${style}，主色调${color}，高清8K，专业商业摄影，锐度高，细节清晰，无水印，无多余文字`;
  if (font !== '无文字') {
    fullPrompt += `，文字字体为${font}，字体清晰，排版美观`;
  }

  // 加载状态
  resultDiv.innerHTML = '<div id="loading">正在批量生成，请稍等...</div>';

  try {
    // 构建请求参数
    const requestData = {
      model: "black-forest-labs/FLUX.1-schnell",
      prompt: fullPrompt,
      negative_prompt: "模糊,水印,文字杂乱,丑陋,变形,低画质,模糊,有杂物,颜色失真",
      image_size: size,
      n: parseInt(count)
    };

    // 如果有参考图，添加图生图参数
    if (refImgBase64) {
      // 提取base64部分（去掉前缀）
      const imgBase64 = refImgBase64.split(',')[1];
      requestData.image = imgBase64;
      requestData.mode = "image-to-image"; // 图生图模式
      requestData.image_strength = 0.7; // 参考图影响程度（0.1-1，越大越像参考图）
    }

    // 调用硅基流动API
    const res = await fetch("https://api.siliconflow.cn/v1/images/generations", {
      method: "POST",
      headers: {
        "Authorization": `Bearer ${API_KEY}`,
        "Content-Type": "application/json"
      },
      body: JSON.stringify(requestData)
    });

    const data = await res.json();

    // 生成成功，展示图片
    if (data?.data?.length) {
      let html = '<div class="download-tip">生成完成，点击图片即可下载</div><div class="result-grid">';
      data.data.forEach(item => {
        html += `<img src="${item.url}" onclick="downloadImg('${item.url}')" title="点击下载">`;
      });
      html += '</div>';
      resultDiv.innerHTML = html;
    } else {
      resultDiv.innerHTML = '生成失败，请重试（可能是API额度不足或描述有误）';
    }
  } catch (err) {
    resultDiv.innerHTML = '生成出错，请检查API密钥是否正确，或刷新页面重试';
    console.log(err);
  }
}

// 下载图片
function downloadImg(url) {
  const a = document.createElement("a");
  a.href = url;
  a.download = `AI详情图_${new Date().getTime()}.png`;
  a.click();
}

// 清空所有设置
function clearAll() {
  document.getElementById('prompt').value = '';
  document.getElementById('refImg').value = '';
  document.getElementById('refPreview').style.display = 'none';
  document.getElementById('size').value = '1024x1536';
  document.getElementById('style').value = '简约风';
  document.getElementById('color').value = '白色';
  document.getElementById('font').value = '无文字';
  document.getElementById('count').value = '1';
  document.getElementById('result').innerHTML = '<p class="download-tip">生成后点击图片即可下载</p>';
  refImgBase64 = null;
}
